224 inline const PointerType *getRefTypeOfHeapAllocOrStatic(const CallSite cs)
225 {
226  const PointerType *refType = NULL;
227  // Case 1: heap object held by *argument, we should get its element type.
228  if (isHeapAllocExtCallViaArg(cs))
229  {
230  int argPos = getHeapAllocHoldingArgPosition(cs);
231  const Value *arg = cs.getArgument(argPos);
232  if (const PointerType *argType = SVFUtil::dyn_cast<PointerType>(arg->getType()))
233  refType = SVFUtil::dyn_cast<PointerType>(argType->getElementType());
234  }
235  // Case 2: heap/static object held by return value.
236  else
237  {
238  assert((isStaticExtCall(cs) || isHeapAllocExtCallViaRet(cs))
239  && "Must be heap alloc via ret, or static allocation site");
240  refType = SVFUtil::dyn_cast<PointerType>(cs.getType());
241  }
242  assert(refType && "Allocated object must be held by a pointer-typed value.");
243  return refType;
244 }
