46  PTASCEV(const Value* p, const SCEV* s, ScalarEvolution* SE): scev(s),start(NULL), step(NULL), ptr(p), inloop(false), tripcount(0)
47  {
48  if(const SCEVAddRecExpr* ar = SVFUtil::dyn_cast<SCEVAddRecExpr>(s))
49  {
50  if (const SCEVConstant *startExpr = SVFUtil::dyn_cast<SCEVConstant>(ar->getStart()))
51  start = startExpr->getValue();
52  if (const SCEVConstant *stepExpr = SVFUtil::dyn_cast<SCEVConstant>(ar->getStepRecurrence(*SE)))
53  step = stepExpr->getValue();
54  tripcount = SE->getSmallConstantTripCount(const_cast<Loop*>(ar->getLoop()));
55  inloop = true;
56  }
57  }

78  inline bool operator< (const PTASCEV& rhs) const
79  {
80  if(start!=rhs.start)
81  return start < rhs.start;
82  else if(step!=rhs.step)
83  return step < rhs.step;
84  else if(ptr!=rhs.ptr)
85  return ptr < rhs.ptr;
86  else if(tripcount!=rhs.tripcount)
87  return tripcount < rhs.tripcount;
88  else
