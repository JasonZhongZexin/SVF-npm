64  virtual NodeStack& SCCDetect()
65  {
67  this->getSCCDetector()->find();

71  NodeStack revTopoStack;
72  NodeStack& topoStack = this->getSCCDetector()->topoNodeStack();
73  while (!topoStack.empty())
74  {
75  NodeID nodeId = topoStack.top();
76  topoStack.pop();
77  const NodeBS& subNodes = this->getSCCDetector()->subNodes(nodeId);
78  for (NodeBS::iterator it = subNodes.begin(), eit = subNodes.end(); it != eit; ++it)
79  {
80  revTopoStack.push(*it);
81  }
82  }

84  assert(nodeStack.empty() && "node stack is not empty, some nodes are not popped properly.");
85 
87  while (!revTopoStack.empty())
88  {
89  NodeID nodeId = revTopoStack.top();
90  revTopoStack.pop();
91  nodeStack.push(nodeId);
92  }

94  return nodeStack;
95  }

104 class WPASCCSolver : public WPAFSSolver<GraphType>

106 public:

108  typedef typename WPASolver<GraphType>::GNODE GNODE;
109  typedef typename WPASolver<GraphType>::child_iterator child_iterator;

112 
113  virtual ~WPASCCSolver() {}
114 
115 protected:
116  virtual void solve()
117  {
120  while (!this->isWorklistEmpty())
121  this->popFromWorklist();
122 
123  NodeStack& nodeStack = this->SCCDetect();
124 
125  while (!nodeStack.empty())
126  {
127  NodeID rep = nodeStack.top();
128  nodeStack.pop();
129 
130  setCurrentSCC(rep);
131 
132  const NodeBS& sccNodes = this->getSCCDetector()->subNodes(rep);
133  for (NodeBS::iterator it = sccNodes.begin(), eit = sccNodes.end(); it != eit; ++it)
134  this->pushIntoWorklist(*it);
135 
136  while (!this->isWorklistEmpty())
137  this->processNode(this->popFromWorklist());
138  }
139  }
140 
142  virtual void propagate(GNODE* v)
143  {
144  child_iterator EI = GTraits::direct_child_begin(v);
145  child_iterator EE = GTraits::direct_child_end(v);
146  for (; EI != EE; ++EI)
147  {
148  if (this->propFromSrcToDst(*(EI.getCurrent())))
149  addNodeIntoWorkList(this->Node_Index(*EI));
150  }
151  }
152 
153  virtual inline void addNodeIntoWorkList(NodeID node)
154  {
155  if (isInCurrentSCC(node))
156  this->pushIntoWorklist(node);
157  }
158 
159  inline bool isInCurrentSCC(NodeID node)
160  {
161  return (const_cast<NodeBS&>(this->getSCCDetector()->subNodes(curSCCID))).test(node);
162  }
163  inline void setCurrentSCC(NodeID id)
164  {

188 protected:
189  virtual void solve()
190  {
191  bool solveAll = true;
194  if (!this->isWorklistEmpty())
195  {
196  solveAll = false;
197  while (!this->isWorklistEmpty())
198  addNewCandidate(this->popFromWorklist());
199  }
200 
201  NodeStack& nodeStack = this->SCCDetect();
202 
203  while (!nodeStack.empty())
204  {
205  NodeID rep = nodeStack.top();
206  nodeStack.pop();
207 
208  this->setCurrentSCC(rep);
209 
210  NodeBS sccNodes = this->getSCCDetector()->subNodes(rep);
211  if (solveAll == false)
212  sccNodes &= getCandidates();
213 
214  for (NodeBS::iterator it = sccNodes.begin(), eit = sccNodes.end(); it != eit; ++it)
215  this->pushIntoWorklist(*it);
216 
217  while (!this->isWorklistEmpty())
218  this->processNode(this->popFromWorklist());
219 
220  removeCandidates(sccNodes);
221  }
222  }
223 
224  virtual inline void addNodeIntoWorkList(NodeID node)
225  {
226  if (this->isInCurrentSCC(node))
227  this->pushIntoWorklist(node);
228  else
229  addNewCandidate(node);
230  }
231 
232 private:
233  inline void addNewCandidate(NodeID node)
234  {
235  candidates.set(node);
236  }
237  inline const NodeBS& getCandidates() const
238  {
239  return candidates;
240  }
241  inline void removeCandidates(const NodeBS& nodes)
242  {
243  candidates.intersectWithComplement(nodes);
244  }
