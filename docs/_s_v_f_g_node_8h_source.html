259 class MSSAPHISVFGNode : public MRSVFGNode
260 {
261 public:
262  typedef DenseMap<u32_t,const MRVer*> OPVers;
263 
264 protected:
265  const MemSSA::MDEF* res;
266  OPVers opVers;
267 
268 public:
270  MSSAPHISVFGNode(NodeID id, const MemSSA::MDEF* def,VFGNodeK k = MPhi): MRSVFGNode(id, k), res(def)
271  {
272  cpts = def->getMR()->getPointsTo();
273  if(const MemSSA::PHI* phi = SVFUtil::dyn_cast<const MemSSA::PHI>(def))
274  bb = phi->getBasicBlock();
275  else if(const MemSSA::ENTRYCHI* enChi = SVFUtil::dyn_cast<const MemSSA::ENTRYCHI>(def))
276  bb = &enChi->getFunction()->getLLVMFun()->getEntryBlock();
277  else if(const MemSSA::CALLCHI* calChi = SVFUtil::dyn_cast<const MemSSA::CALLCHI>(def))
278  bb = calChi->getBasicBlock();
279  else
280  assert("what else def for MSSAPHI node?");
281  }
283 
284  inline const MRVer* getOpVer(u32_t pos) const
285  {
286  OPVers::const_iterator it = opVers.find(pos);
287  assert(it!=opVers.end() && "version is NULL, did not rename?");
288  return it->second;
289  }
290  inline void setOpVer(u32_t pos, const MRVer* node)
291  {
292  opVers[pos] = node;
293  }
294  inline const MemSSA::MDEF* getRes() const
295  {
296  return res;
297  }
298  inline u32_t getOpVerNum() const

300  return opVers.size();

303  {
