135  static inline GEdgeFlag makeEdgeFlagWithCallInst(GEdgeKind k, const ICFGNode* cs)
136  {
137  Inst2LabelMap::const_iterator iter = inst2LabelMap.find(cs);
138  u64_t label = (iter != inst2LabelMap.end()) ?
139  iter->second : callEdgeLabelCounter++;
140  return (label << EdgeKindMaskBits) | k;
141  }
142 
145  static inline GEdgeFlag makeEdgeFlagWithStoreInst(GEdgeKind k, const ICFGNode* store)
146  {
147  Inst2LabelMap::const_iterator iter = inst2LabelMap.find(store);
148  u64_t label = (iter != inst2LabelMap.end()) ?
149  iter->second : storeEdgeLabelCounter++;
150  return (label << EdgeKindMaskBits) | k;
151  }
