84 /*
85  * Location set represents a set of locations in a memory block with following offsets:
86  * { offset + \sum_{i=0}^N (stride_i * j_i) | 0 \leq j_i < M_i }
87  * where N is the size of number-stride pair vector, M_i (stride_i) is i-th number (stride)
88  * in the number-stride pair vector.
89  */

150  inline bool operator< (const LocationSet& rhs) const
151  {
152  if (fldIdx != rhs.fldIdx)
153  return (fldIdx < rhs.fldIdx);
154 // else if (byteOffset != rhs.byteOffset)
155 // return (byteOffset < rhs.byteOffset);
156  else
157  {
158  const ElemNumStridePairVec& pairVec = getNumStridePair();
159  const ElemNumStridePairVec& rhsPairVec = rhs.getNumStridePair();
160  if (pairVec.size() != rhsPairVec.size())
161  return (pairVec.size() < rhsPairVec.size());
162  else
163  {
164  ElemNumStridePairVec::const_iterator it = pairVec.begin();
165  ElemNumStridePairVec::const_iterator rhsIt = rhsPairVec.begin();
166  for (; it != pairVec.end() && rhsIt != rhsPairVec.end(); ++it, ++rhsIt)
167  {
168  if ((*it).first != (*rhsIt).first)
169  return ((*it).first < (*rhsIt).first);
170  else if ((*it).second != (*rhsIt).second)
171  return ((*it).second < (*rhsIt).second);
172  }
173 
174  return false;
175  }
176  }

219  static inline LSRelation checkRelation(const LocationSet& LHS, const LocationSet& RHS)
220  {
221  PointsTo lhsLocations = LHS.computeAllLocations();
222  PointsTo rhsLocations = RHS.computeAllLocations();
223  if (lhsLocations.intersects(rhsLocations))
224  {
225  if (lhsLocations == rhsLocations)
226  return Same;
227  else if (lhsLocations.contains(rhsLocations))
228  return Superset;
229  else if (rhsLocations.contains(lhsLocations))
230  return Subset;
231  else
232  return Overlap;
233  }
234  else
235  {
236  return NonOverlap;
237  }
