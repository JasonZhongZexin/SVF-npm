254  inline bool setCFCond(const BasicBlock* bb, Condition* cond)
255  {
256  BBToCondMap::iterator it = bbToCondMap.find(bb);
257  if(it!=bbToCondMap.end() && it->second == cond)
258  return false;
259 
260  bbToCondMap[bb] = cond;
261  return true;
262  }
263  inline Condition* getCFCond(const BasicBlock* bb) const
264  {
265  BBToCondMap::const_iterator it = bbToCondMap.find(bb);
266  if(it==bbToCondMap.end())
267  {
268  return getFalseCond();
269  }
270  return it->second;
271  }
