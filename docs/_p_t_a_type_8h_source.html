134  // Algorithm set_intersection
135  // Complexity: 2 * (N1 + N2) - 1
136  // N1, N2: number of element in the two typeset
137  inline bool intersect(const TypeSet *typeset) const
138  {
139  if (size() == 1)
140  {
141  const_iterator first = begin();
142  return typeset->containType(*first);
143  }
144  else if (typeset->size() == 1)
145  {
146  const_iterator first = typeset->begin();
147  return containType(*first);
148  }
149  else
150  {
151  const_iterator first1 = typeset->begin(), first2 = begin(),
152  last1 = typeset->end(), last2 = end();
153  const_iterator largest1 = last1, largest2 = last2;
154  largest1--;
155  largest2--;
156  if (*largest1 < *first2 || *largest2 < *first1)
157  return false;
158 
159  while (first1 != last1 && first2 != last2)
160  {
161  if (*first1 < *first2)
162  first1++;
163  else if (*first2 < *first1)
164  first2++;
165  else
166  return true;
167  }
168  return false;
169  }
170  }

318  /*
319  * Translate llvm type into ptatype and build the pagnode to ptatype map
320  *
321  * Kinds of PAGNode:
322  * ValPN: GepValPN, DummyValPN
323  * ObjPN: GepObjPN, FIObjPN, DummyObjPN
324  * RetPN
325  * VarArgPN
326  */
327  void translateLLVMTypeToPTAType(const PAG *pag)
328  {
329  for (PAG::const_iterator it = pag->begin(); it != pag->end(); ++it)
330  {
331  const PAGNode *pagNode = it->second;
332  if (pagNode->hasValue() == false)
333  continue;
334 
335  const Value *value = pagNode->getValue();
336  const Type *valType = value->getType();
337 
338  const Type *nodeType = valType;
339 
340  if (const GepValPN *gepvalnode = SVFUtil::dyn_cast<GepValPN>(pagNode))
341  {
342  nodeType = gepvalnode->getType();
343  }
344  else if (SVFUtil::isa<RetPN>(pagNode))
345  {
346  const llvm::PointerType *ptrTy = SVFUtil::dyn_cast<llvm::PointerType>(valType);
347  const llvm::FunctionType *funTy = SVFUtil::dyn_cast<llvm::FunctionType>(ptrTy->getElementType());
348  nodeType = funTy->getReturnType();
349  }
350 
351  PTAType ptaType(nodeType);
352  if (addTypeForVar(pagNode->getId(), ptaType))
353  addVarForType(pagNode->getId(), ptaType);
354  }
355  }
