102  void countStat()
103  {
104  ICFG::ICFGNodeIDToNodeMapTy::iterator it = icfg->begin();
105  ICFG::ICFGNodeIDToNodeMapTy::iterator eit = icfg->end();
106  for (; it != eit; ++it)
107  {
108  numOfNodes++;
109 
110  ICFGNode *node = it->second;
111 
112  if (SVFUtil::isa<IntraBlockNode>(node))
113  numOfIntraNodes++;
114  else if (SVFUtil::isa<CallBlockNode>(node))
115  numOfCallNodes++;
116  else if (SVFUtil::isa<RetBlockNode>(node))
117  numOfRetNodes++;
118  else if (SVFUtil::isa<FunEntryBlockNode>(node))
119  numOfEntryNodes++;
120  else if (SVFUtil::isa<FunExitBlockNode>(node))
121  numOfExitNodes++;
122 
123 
124  ICFGEdge::ICFGEdgeSetTy::iterator edgeIt =
125  it->second->OutEdgeBegin();
126  ICFGEdge::ICFGEdgeSetTy::iterator edgeEit =
127  it->second->OutEdgeEnd();
128  for (; edgeIt != edgeEit; ++edgeIt)
129  {
130  const ICFGEdge *edge = *edgeIt;
131  numOfEdges++;
132  if (edge->isCallCFGEdge())
133  numOfCallEdges++;
134  else if (edge->isRetCFGEdge())
135  numOfRetEdges++;
136  else if (edge->isIntraCFGEdge())
137  numOfIntraEdges++;
138  }
139  }
140  }
