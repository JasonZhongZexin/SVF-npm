223  inline const Value* getJoinedThread(const Instruction *inst) const
226  {
227  assert(isTDJoin(inst) && "not a thread join function!");
228  CallSite cs = getLLVMCallSite(inst);
229  Value* join = cs.getArgument(0);
230  if(SVFUtil::isa<LoadInst>(join))
231  return SVFUtil::cast<LoadInst>(join)->getPointerOperand();
232  else if(SVFUtil::isa<Argument>(join))
233  return join;
234  assert(false && "the value of the first argument at join is not a load instruction?");
235  return NULL;
236  }
