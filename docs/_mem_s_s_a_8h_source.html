234  {

244  inline void RenameChiSet(const CHISet& chiSet, MRVector& memRegs)
245  {
246  for (CHISet::iterator cit = chiSet.begin(), ecit = chiSet.end();
247  cit != ecit; ++cit)
248  {
249  CHI* chi = (*cit);
250  chi->setOpVer(getTopStackVer(chi->getMR()));
251  chi->setResVer(newSSAName(chi->getMR(),chi));
252  memRegs.push_back(chi->getMR());
253  }
