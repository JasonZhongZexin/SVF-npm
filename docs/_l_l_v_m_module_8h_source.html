192  bool allCTir(void) const
193  {
194  // Iterate over all modules. If a single module does not have the correct ctir module flag,
195  // short-circuit and return false.
196  for (u32_t i = 0; i < getModuleNum(); ++i)
197  {
198  llvm::Metadata *ctirModuleFlag = llvmModuleSet->getModule(i)->getModuleFlag(cppUtil::ctir::derefMDName);
199  if (ctirModuleFlag == nullptr)
200  {
201  return false;
202  }
203 
204  llvm::ConstantAsMetadata *flagConstMetadata = SVFUtil::dyn_cast<llvm::ConstantAsMetadata>(ctirModuleFlag);
205  ConstantInt *flagConstInt = SVFUtil::dyn_cast<ConstantInt>(flagConstMetadata->getValue());
206  if (flagConstInt->getZExtValue() != cppUtil::ctir::moduleFlagValue)
207  {
208  return false;
209  }
210  }
211 
212  return true;
213  }
