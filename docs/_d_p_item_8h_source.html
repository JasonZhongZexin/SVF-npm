243  inline virtual bool pushContext(NodeID ctx)
244  {
245 
246  if(context.size() < maximumCxtLen)
247  {
248  context.push_back(ctx);
249 
250  if(context.size() > maximumCxt)
251  maximumCxt = context.size();
252  return true;
253  }
254  else
255  {
256  if(!context.empty())
257  {
258  setNonConcreteCxt();
259  context.erase(context.begin());
260  context.push_back(ctx);
261  }
262  return false;
263  }
264  }

268  {
270  if(context.empty())
271  return true;

276  return true;
277  }
278  return false;
279  }

489  {
490  // assert(!hasVFEdge(from,to) && "Edge exit?");
491  if(edges.size() > maximumPath)
492  maximumPath = edges.size();
493 
494  edges.push_back(std::make_pair(from,to));
495  }
497  inline bool hasVFEdge(NodeID from, NodeID to) const
498  {
499  return std::find(edges.begin(),edges.end(),std::make_pair(from,to)) != edges.end();
500  }
502  inline bool hasIncomingEdge(NodeID node) const
503  {
504  for(EdgeSet::const_iterator it = edges.begin(), eit =edges.end(); it!=eit; ++it)
505  {
506  if(it->second == node)
507  return true;
508  }
509  return false;
510  }
512  inline bool hasOutgoingEdge(NodeID node) const
513  {
514  for(EdgeSet::const_iterator it = edges.begin(), eit =edges.end(); it!=eit; ++it)
515  {
516  if(it->first == node)
517  return true;
518  }
519  return false;
520  }
521  inline bool addPath(PathCondAllocator* allocator, PathCond* c, NodeID from, NodeID to)
522  {
523  if(pathLen() < maximumPathLen)
524  {
525  if(!hasVFEdge(from,to))
526  {
528  if(hasOutgoingEdge(from))
529  c = allocator->getTrueCond();
530  }
531  addVFEdge(from,to);
532  return condAnd(allocator,c);
533  }
534  // DBOUT(DDDA, SVFUtil::outs() << "\t\t!!path length beyond limits \n");
535  return true;
536  }
538 
539  inline bool condAnd(PathCondAllocator* allocator, PathCond* c)
540  {

542  return path != allocator->getFalseCond();
543  }
