138  virtual void handleSingleStatement(const DPIm& dpm, CPtSet& pts)
139  {
141  resolveFunPtr(dpm);
142 
143  const SVFGNode* node = dpm.getLoc();
144  if(SVFUtil::isa<AddrSVFGNode>(node))
145  {
146  handleAddr(pts,dpm,SVFUtil::cast<AddrSVFGNode>(node));
147  }
148  else if(SVFUtil::isa<CopySVFGNode>(node) || SVFUtil::isa<PHISVFGNode>(node)
149  || SVFUtil::isa<ActualParmSVFGNode>(node) || SVFUtil::isa<FormalParmSVFGNode>(node)
150  || SVFUtil::isa<ActualRetSVFGNode>(node) || SVFUtil::isa<FormalRetSVFGNode>(node)
151  || SVFUtil::isa<NullPtrSVFGNode>(node))
152  {
153  backtraceAlongDirectVF(pts,dpm);
154  }
155  else if(SVFUtil::isa<GepSVFGNode>(node))
156  {
157  CPtSet gepPts;
158  backtraceAlongDirectVF(gepPts,dpm);
159  unionDDAPts(pts, processGepPts(SVFUtil::cast<GepSVFGNode>(node),gepPts));
160  }
161  else if(SVFUtil::isa<LoadSVFGNode>(node))
162  {
163  CPtSet loadpts;
164  startNewPTCompFromLoadSrc(loadpts,dpm);
165  for(typename CPtSet::iterator it = loadpts.begin(), eit = loadpts.end(); it!=eit; ++it)
166  {
167  backtraceAlongIndirectVF(pts,getDPImWithOldCond(dpm,*it,node));
168  }
169  }
170  else if(SVFUtil::isa<StoreSVFGNode>(node))
171  {
172  if(isMustAlias(getLoadDpm(dpm),dpm))
173  {
174  DBOUT(DDDA, SVFUtil::outs() << "+++must alias for load and store:");
175  DBOUT(DDDA, getLoadDpm(dpm).dump());
176  DBOUT(DDDA, dpm.dump());
177  DBOUT(DDDA, SVFUtil::outs() << "+++\n");
178  DOSTAT(ddaStat->_NumOfMustAliases++);
179  backtraceToStoreSrc(pts,dpm);
180  }
181  else
182  {
183  CPtSet storepts;
184  startNewPTCompFromStoreDst(storepts,dpm);
185  for(typename CPtSet::iterator it = storepts.begin(), eit = storepts.end(); it!=eit; ++it)
186  {
187  if(propagateViaObj(*it,getLoadCVar(dpm)))
188  {
189  backtraceToStoreSrc(pts,getDPImWithOldCond(dpm,*it,node));
190 
191  if(isStrongUpdate(storepts,SVFUtil::cast<StoreSVFGNode>(node)))
192  {
193  DBOUT(DDDA, SVFUtil::outs() << "backward strong update for obj " << dpm.getCurNodeID() << "\n");
194  DOSTAT(addSUStat(dpm,node);)
195  }
196  else
197  {
198  DOSTAT(rmSUStat(dpm,node);)
199  backtraceAlongIndirectVF(pts,getDPImWithOldCond(dpm,*it,node));
200  }
201  }
202  else
203  {
204  backtraceAlongIndirectVF(pts,dpm);
205  }
206  }
207  }
208  }
209  else if(SVFUtil::isa<MRSVFGNode>(node))
210  {
211  backtraceAlongIndirectVF(pts,dpm);
212  }
213  else
214  assert(false && "unexpected kind of SVFG nodes");
215  }

218  void reCompute(const DPIm& dpm)
219  {
221  SVFGEdgeSet newIndirectEdges;
222  if(_pag->isFunPtr(dpm.getCurNodeID()))
223  {
224  const CallSiteSet& csSet = _pag->getIndCallSites(dpm.getCurNodeID());
225  for(CallSiteSet::const_iterator it = csSet.begin(), eit = csSet.end(); it!=eit; ++it)
226  updateCallGraphAndSVFG(dpm, (*it),newIndirectEdges);

230  _callGraphSCC->find();
231  reComputeForEdges(dpm,newIndirectEdges,true);

235  reComputeForEdges(dpm,edgeSet,false);

239  void reComputeForEdges(const DPIm& dpm, const SVFGEdgeSet& edgeSet, bool indirectCall = false)
240  {
241  for (SVFGNode::const_iterator it = edgeSet.begin(), eit = edgeSet.end(); it != eit; ++it)
242  {
243  const SVFGEdge* edge = *it;
244  const SVFGNode* dst = edge->getDstNode();
245  typename LocToDPMVecMap::const_iterator locIt = getLocToDPMVecMap().find(dst->getId());
247  if (locIt == getLocToDPMVecMap().end())
248  continue;
249  DPTItemSet dpmSet(locIt->second.begin(), locIt->second.end());
250  for(typename DPTItemSet::const_iterator it = dpmSet.begin(),eit = dpmSet.end(); it!=eit; ++it)
251  {
252  const DPIm& dstDpm = *it;
253  if(!indirectCall && SVFUtil::isa<IndirectSVFGEdge>(edge) && !SVFUtil::isa<LoadSVFGNode>(edge->getDstNode()))
254  {
255  if(dstDpm.getCurNodeID() == dpm.getCurNodeID())
256  {
257  DBOUT(DDDA,SVFUtil::outs() << "\t Recompute, forward from :");
258  DBOUT(DDDA, dpm.dump());
259  DOSTAT(ddaStat->_NumOfStepInCycle++);
260  clearbkVisited(dstDpm);
261  findPT(dstDpm);
262  }
263  }
264  else
265  {
266  if(indirectCall)
267  DBOUT(DDDA,SVFUtil::outs() << "\t Recompute for indirect call from :");
268  else
269  DBOUT(DDDA,SVFUtil::outs() << "\t Recompute forward from :");
270  DBOUT(DDDA, dpm.dump());
271  DOSTAT(ddaStat->_NumOfStepInCycle++);
272  clearbkVisited(dstDpm);
273  findPT(dstDpm);
274  }
275  }
276  }
