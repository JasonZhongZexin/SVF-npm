298  inline bool mustAlias(const CVar& var1, const CVar& var2)
299  {
300  if(isSameVar(var1,var2))
301  return true;
302 
303  bool singleton = !(isHeapMemObj(var1.get_id()) || isLocalVarInRecursiveFun(var1.get_id()));
304  if(isCondCompatible(var1.get_cond(),var2.get_cond(),singleton) == false)
305  return false;
306 
307  const CPtSet& cpts1 = getPts(var1);
308  const CPtSet& cpts2 = getPts(var2);
309  return (contains(cpts1,cpts2) && contains(cpts2,cpts1));
310  }
311 
312  // Whether cpts1 contains all points-to targets of pts2
313  bool contains(const CPtSet& cpts1, const CPtSet& cpts2)
314  {
315  if (cpts1.empty() || cpts2.empty())
316  return false;
317 
318  for (typename CPtSet::const_iterator it2 = cpts2.begin(); it2 != cpts2.end(); ++it2)
319  {
320  bool hasObj = false;
321  for (typename CPtSet::const_iterator it1 = cpts1.begin(); it1 != cpts1.end(); ++it1)
322  {
323  if(isSameVar(*it1,*it2))
324  {
325  hasObj = true;
326  break;
327  }
328  }
329  if(hasObj == false)
330  return false;
331  }
332  return true;
333  }

336  bool isSameVar(const CVar& var1, const CVar& var2) const
337  {
338  if(var1.get_id() != var2.get_id())
339  return false;
340 
342  bool singleton = !(isHeapMemObj(var1.get_id()) || isLocalVarInRecursiveFun(var1.get_id()));
343  return isCondCompatible(var1.get_cond(),var2.get_cond(),singleton);
344  }

428  {
429  CPtSet cpts1;

431  CPtSet cpts2;
432  expandFIObjs(pts2,cpts2);
433  if (containBlackHoleNode(cpts1) || containBlackHoleNode(cpts2))
434  return llvm::MayAlias;
