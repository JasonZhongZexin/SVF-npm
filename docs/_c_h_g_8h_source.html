218  inline s32_t getVirtualFunctionID(const SVFFunction* vfn) const
219  {
220  std::map<const SVFFunction*, s32_t>::const_iterator it =
221  virtualFunctionToIDMap.find(vfn);
222  if (it != virtualFunctionToIDMap.end())
223  return it->second;
224  else
225  return -1;
226  }
227  inline const SVFFunction* getVirtualFunctionBasedonID(s32_t id) const
228  {
229  std::map<const SVFFunction*, s32_t>::const_iterator it, eit;
230  for (it = virtualFunctionToIDMap.begin(), eit =
231  virtualFunctionToIDMap.end(); it != eit; ++it)
232  {
233  if (it->second == id)
234  return it->first;
235  }
236  return NULL;
237  }
