40 public:
42  CxtStmt(const CallStrCxt& c, const Instruction* f) :cxt(c), inst(f)

44  }
46  CxtStmt(const CxtStmt& ctm) : cxt(ctm.getContext()),inst(ctm.getStmt())
47  {
48  }
50  virtual ~CxtStmt()
51  {
52  }
54  inline const CallStrCxt& getContext() const
55  {
56  return cxt;
57  }
59  inline const Instruction* getStmt() const
60  {
61  return inst;
62  }
65  inline bool operator< (const CxtStmt& rhs) const
66  {
67  if(inst!=rhs.getStmt())
68  return inst < rhs.getStmt();
69  else
70  return cxt < rhs.getContext();
71  }
73  inline CxtStmt& operator= (const CxtStmt& rhs)
74  {
75  if(*this!=rhs)
76  {
77  inst = rhs.getStmt();
78  cxt = rhs.getContext();
79  }
80  return *this;
81  }

84  {
85  return (inst == rhs.getStmt() && cxt == rhs.getContext());
86  }
88  inline bool operator!= (const CxtStmt& rhs) const

91  }

104  }
