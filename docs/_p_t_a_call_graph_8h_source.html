303  inline CallSiteID addCallSite(const CallBlockNode* cs, const SVFFunction* callee)
304  {
305  std::pair<const CallBlockNode*, const SVFFunction*> newCS(std::make_pair(cs, callee));
306  CallSiteToIdMap::const_iterator it = csToIdMap.find(newCS);
307  //assert(it == csToIdMap.end() && "cannot add a callsite twice");
308  if(it == csToIdMap.end())
309  {
310  CallSiteID id = totalCallSiteNum++;
311  csToIdMap.insert(std::make_pair(newCS, id));
312  idToCSMap.insert(std::make_pair(id, newCS));
313  return id;
314  }
315  return it->second;
316  }
