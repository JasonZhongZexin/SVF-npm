42 class ExtAPI
43 {
44 public:
45  //External Function types
46  //Assume a call in the form LHS= F(arg0, arg1, arg2, arg3).
47  enum extf_t
48  {
49  EFT_NOOP= 0, //no effect on pointers
50  EFT_ALLOC, //returns a ptr to a newly allocated object
51  EFT_REALLOC, //like L_A0 if arg0 is a non-null ptr, else ALLOC
52  EFT_FREE, //free memory arg0 and all pointers passing into free function
53  EFT_NOSTRUCT_ALLOC, //like ALLOC but only allocates non-struct data
54  EFT_STAT, //retval points to an unknown static var X
55  EFT_STAT2, //ret -> X -> Y (X, Y - external static vars)
56  EFT_L_A0, //copies arg0, arg1, or arg2 into LHS
57  EFT_L_A1,
58  EFT_L_A2,
59  EFT_L_A8,
60  EFT_L_A0__A0R_A1R, //copies the data that arg1 points to into the location
61  // arg0 points to; note that several fields may be
62  // copied at once if both point to structs.
63  // Returns arg0.
64  EFT_A1R_A0R, //copies *arg0 into *arg1, with non-ptr return
65  EFT_A3R_A1R_NS, //copies *arg1 into *arg3 (non-struct copy only)
66  EFT_A1R_A0, //stores arg0 into *arg1
67  EFT_A2R_A1, //stores arg1 into *arg2
68  EFT_A4R_A1, //stores arg1 into *arg4
69  EFT_L_A0__A2R_A0, //stores arg0 into *arg2 and returns it
70  EFT_A0R_NEW, //stores a pointer to an allocated object in *arg0
71  EFT_A1R_NEW, //as above, into *arg1, etc.
72  EFT_A2R_NEW,
73  EFT_A4R_NEW,
74  EFT_A11R_NEW,
75  EFT_STD_RB_TREE_INSERT_AND_REBALANCE, // Some complex effects
76  EFT_STD_RB_TREE_INCREMENT, // Some complex effects
77  EFT_STD_LIST_HOOK, // Some complex effects
78 
79  CPP_EFT_A0R_A1, //stores arg1 into *arg0
80  CPP_EFT_A0R_A1R, //copies *arg1 into *arg0
81  CPP_EFT_A1R, //load arg1
82  EFT_CXA_BEGIN_CATCH, //__cxa_begin_catch
83  CPP_EFT_DYNAMIC_CAST, // dynamic_cast
84  EFT_OTHER //not found in the list
85  };

88  //Each Function name is mapped to its extf_t

92  std::map<const SVFFunction*, bool> isext_cache;

117  //Return the extf_t of (F).
118  extf_t get_type(const SVFFunction* F) const
119  {
120  assert(F);
121  std::string funName = F->getName().str();
122  if(F->isIntrinsic())
123  {
124  funName = "llvm." + F->getName().split('.').second.split('.').first.str();
125  }
126  llvm::StringMap<extf_t>::const_iterator it= info.find(funName);
127  if(it == info.end() || !F->isDeclaration())
128  return EFT_OTHER;
129  else
130  return it->second;
131  }
132 
133  //Does (F) have a static var X (unavailable to us) that its return points to?
134  bool has_static(const SVFFunction* F) const
135  {
136  extf_t t= get_type(F);
137  return t==EFT_STAT || t==EFT_STAT2;
138  }
139  //Assuming hasStatic(F), does (F) have a second static Y where X -> Y?
140  bool has_static2(const SVFFunction* F) const
141  {
142  return get_type(F) == EFT_STAT2;
143  }
144  //Does (F) allocate a new object and return it?
145  bool is_alloc(const SVFFunction* F) const
146  {
147  extf_t t= get_type(F);
148  return t==EFT_ALLOC || t==EFT_NOSTRUCT_ALLOC;
149  }
150  //Does (F) allocate a new object and assign it to one of its arguments?
151  bool is_arg_alloc(const SVFFunction* F) const
152  {
153  extf_t t= get_type(F);
154  return t>=EFT_A0R_NEW && t<=EFT_A11R_NEW;
155  }
156  //Get the position of argument which holds the new object
157  int get_alloc_arg_pos(const SVFFunction* F) const
158  {
159  extf_t t= get_type(F);
160  switch(t)
161  {
162  case EFT_A0R_NEW:
163  return 0;
164  case EFT_A1R_NEW:
165  return 1;
166  case EFT_A2R_NEW:
167  return 2;
168  case EFT_A4R_NEW:
169  return 4;
170  case EFT_A11R_NEW:
171  return 11;
172  default:
173  assert(false && "Not an alloc call via argument.");
174  return -1;
175  }
176  }
177  //Does (F) allocate only non-struct objects?
178  bool no_struct_alloc(const SVFFunction* F) const
179  {
180  return get_type(F) == EFT_NOSTRUCT_ALLOC;
181  }
182  //Does (F) not free/release any memory?
183  bool is_dealloc(const SVFFunction* F) const
184  {
185  extf_t t= get_type(F);
186  return t == EFT_FREE;
187  }
188  //Does (F) not do anything with the known pointers?
189  bool is_noop(const SVFFunction* F) const
190  {
191  extf_t t= get_type(F);
192  return t == EFT_NOOP || t == EFT_FREE;
193  }
194  //Does (F) reallocate a new object?
195  bool is_realloc(const SVFFunction* F) const
196  {
197  extf_t t= get_type(F);
198  return t==EFT_REALLOC;
199  }
200  //Should (F) be considered "external" (either not defined in the program
201  // or a user-defined version of a known alloc or no-op)?
202  bool is_ext(const SVFFunction* F)
203  {
204  assert(F);
205  //Check the cache first; everything below is slower.
206  std::map<const SVFFunction*, bool>::iterator i_iec= isext_cache.find(F);
207  if(i_iec != isext_cache.end())
208  return i_iec->second;
209 
210  bool res;
211  if(F->isDeclaration() || F->isIntrinsic())
212  {
213  res= 1;
214  }
215  else
216  {
217  extf_t t= get_type(F);
218  res= t==EFT_ALLOC || t==EFT_REALLOC || t==EFT_NOSTRUCT_ALLOC
219  || t==EFT_NOOP || t==EFT_FREE;
220  }
221  isext_cache[F]= res;
222  return res;
223  }
224 };
