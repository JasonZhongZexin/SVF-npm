124  inline bool hasEdge(ConstraintNode* src, ConstraintNode* dst, ConstraintEdge::ConstraintEdgeK kind)
125  {
126  ConstraintEdge edge(src,dst,kind);
127  if(kind == ConstraintEdge::Copy ||
128  kind == ConstraintEdge::NormalGep || kind == ConstraintEdge::VariantGep)
129  return directEdgeSet.find(&edge) != directEdgeSet.end();
130  else if(kind == ConstraintEdge::Addr)
131  return AddrCGEdgeSet.find(&edge) != AddrCGEdgeSet.end();
132  else if(kind == ConstraintEdge::Store)
133  return StoreCGEdgeSet.find(&edge) != StoreCGEdgeSet.end();
134  else if(kind == ConstraintEdge::Load)
135  return LoadCGEdgeSet.find(&edge) != LoadCGEdgeSet.end();
136  else
137  assert(false && "no other kind!");
138  return false;
139  }

142  inline ConstraintEdge* getEdge(ConstraintNode* src, ConstraintNode* dst, ConstraintEdge::ConstraintEdgeK kind)

145  if(kind == ConstraintEdge::Copy || kind == ConstraintEdge::NormalGep || kind == ConstraintEdge::VariantGep)
146  {
