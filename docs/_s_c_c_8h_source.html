147  inline bool isInCycle(NodeID n) const
148  {
149  NodeID rep = repNode(n);
150  // multi-node cycle
151  if (subNodes(rep).count() > 1)
152  {
153  return true;

247  void visit(NodeID v)
248  {
249  // SVFUtil::outs() << "visit GNODE: " << Node_Index(v)<< "\n";
250  _I += 1;
251  _D[v] = _I;
252  this->rep(v,v);
253  this->setVisited(v,true);
254 
255  child_iterator EI = GTraits::direct_child_begin(Node(v));
256  child_iterator EE = GTraits::direct_child_end(Node(v));
257 
258  for (; EI != EE; ++EI)
259  {
260  NodeID w = Node_Index(*EI);
261 
262  if (!this->visited(w))
263  visit(w);
264  if (!this->inSCC(w))
265  {
266  NodeID rep;
267  rep = _D[this->rep(v)] < _D[this->rep(w)] ?
268  this->rep(v) : this->rep(w);
269  this->rep(v,rep);
270  }
271  }
272  if (this->rep(v) == v)
273  {
274  this->setInSCC(v,true);
275  while (!_SS.empty())
276  {
277  NodeID w = _SS.top();
278  if (_D[w] <= _D[v])
279  break;
280  else
281  {
282  _SS.pop();
283  this->setInSCC(w,true);
284  this->rep(w,v);
285  }
286  }
287  _T.push(v);
288  }
289  else
290  _SS.push(v);
291  }

306  void find(void)
307  {
308  // Visit each unvisited root node. A root node is defined
309  // to be a node that has no incoming copy/skew edges
310  clear();
311  node_iterator I = GTraits::nodes_begin(_graph);
312  node_iterator E = GTraits::nodes_end(_graph);
313  for (; I != E; ++I)
314  {
315  NodeID node = Node_Index(*I);
316  if (!this->visited(node))
317  {
318  // We skip any nodes that have a representative other than
319  // themselves. Such nodes occur as a result of merging
320  // nodes either through unifying an ACC or other node
321  // merging optimizations. Any such node should have no
322  // outgoing edges and therefore should no longer be a member
323  // of an SCC.
324  if (this->rep(node) == UINT_MAX || this->rep(node) == node)
325  visit(node);
326  else
327  this->visited(node);
328  }
329  }
330  }
331 
332  void find(NodeSet &candidates)
333  {
334  // This function is reloaded to only visit candidate NODES
335  clear();
336  for (NodeID node : candidates)
337  {
338  if (!this->visited(node))
339  {
340  if (this->rep(node) == UINT_MAX || this->rep(node) == node)
341  visit(node);
342  else
343  this->visited(node);
344  }
345  }
346  }
