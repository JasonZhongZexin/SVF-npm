86  virtual inline void connectAParamAndFParam(const PAGNode* cs_arg, const PAGNode* fun_arg, const CallBlockNode* cs, CallSiteID csId, SVFGEdgeSetTy& edges)
87  {
88  NodeID phiId = getDef(fun_arg);
89  SVFGEdge* edge = addCallEdge(getDef(cs_arg), phiId, csId);
90  if (edge != NULL)
91  {
92  PHISVFGNode* phi = SVFUtil::cast<PHISVFGNode>(getSVFGNode(phiId));

110  virtual inline void connectAInAndFIn(const ActualINSVFGNode* actualIn, const FormalINSVFGNode* formalIn, CallSiteID csId, SVFGEdgeSetTy& edges)
111  {
112  PointsTo intersection = actualIn->getPointsTo();
113  intersection &= formalIn->getPointsTo();
114  if (intersection.empty() == false)
115  {
116  NodeID aiDef = getActualINDef(actualIn->getId());
117  SVFGEdge* edge = addCallIndirectSVFGEdge(aiDef,formalIn->getId(),csId,intersection);
118  if (edge != NULL)
119  edges.insert(edge);
120  }
121  }
123  virtual inline void connectFOutAndAOut(const FormalOUTSVFGNode* formalOut, const ActualOUTSVFGNode* actualOut, CallSiteID csId, SVFGEdgeSetTy& edges)
124  {
125  PointsTo intersection = formalOut->getPointsTo();
126  intersection &= actualOut->getPointsTo();
127  if (intersection.empty() == false)
128  {
129  NodeID foDef = getFormalOUTDef(formalOut->getId());
130  SVFGEdge* edge = addRetIndirectSVFGEdge(foDef,actualOut->getId(),csId,intersection);
131  if (edge != NULL)
132  edges.insert(edge);
133  }
134  }
