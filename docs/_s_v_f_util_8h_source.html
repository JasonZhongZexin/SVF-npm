98 inline bool cmpPts (const PointsTo& lpts,const PointsTo& rpts)
99 {
100  if (lpts.count() != rpts.count())
101  return (lpts.count() < rpts.count());
102  else
103  {
104  PointsTo::iterator bit = lpts.begin(), eit = lpts.end();
105  PointsTo::iterator rbit = rpts.begin(), reit = rpts.end();
106  for (; bit != eit && rbit != reit; bit++, rbit++)
107  {
108  if (*bit != *rbit)
109  return (*bit < *rbit);
110  }
111 
112  return false;
113  }
114 }

158 inline const SVFFunction* getFunction(StringRef name)
159 {

161  LLVMModuleSet* llvmModuleset = LLVMModuleSet::getLLVMModuleSet();
162 
163  for (u32_t i = 0; i < llvmModuleset->getModuleNum(); ++i)

165  Module *mod = llvmModuleset->getModule(i);
166  fun = mod->getFunction(name);

169  return llvmModuleset->getSVFFunction(fun);
170  }
171  }
