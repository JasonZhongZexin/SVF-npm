39 // Define a template that can be specialized by smart pointers to reflect the
40 // fact that they are automatically dereferenced, and are not involved with the
41 // template selection process... the default implementation is a noop.
42 //
43 template<typename From> struct simplify_type
44 {
45  using SimpleType = From; // The real type this represents...
46 
47  // An accessor to get the real value...
48  static SimpleType &getSimplifiedValue(From &Val)
49  {
50  return Val;
51  }
52 };
53 
54 template<typename From> struct simplify_type<const From>
55 {
56  using NonConstSimpleType = typename simplify_type<From>::SimpleType;
57  using SimpleType =
58  typename llvm::add_const_past_pointer<NonConstSimpleType>::type;
59  using RetType =
60  typename llvm::add_lvalue_reference_if_not_pointer<SimpleType>::type;
61 
62  static RetType getSimplifiedValue(const From& Val)
63  {
64  return simplify_type<From>::getSimplifiedValue(const_cast<From&>(Val));
65  }
66 };

68 // The core of the implementation of llvm::isa<X> is here; To and From should be
69 // the names of classes. This template can be specialized to customize the
70 // implementation of llvm::isa<> without rewriting it from scratch.
71 template <typename To, typename From, typename Enabler = void>
72 struct isa_impl
73 {
74  static inline bool doit(const From &Val)
75  {
76  return To::classof(&Val);
77  }

284 // cast<X> - Return the argument parameter cast to the specified type. This
285 // casting operator asserts that the type is correct, so it does not return null
286 // on failure. It does not allow a null argument (use cast_or_null for that).
287 // It is typically used like this:
288 //
289 // cast<Instruction>(myVal)->getParent()
290 //
291 template <class X, class Y>
292 inline typename std::enable_if<!is_simple_type<Y>::value,
293  typename cast_retty<X, const Y>::ret_type>::type
294  cast(const Y &Val)
295 {
296  assert(llvm::isa<X>(Val) && "cast<Ty>() argument of incompatible type!");
297  return cast_convert_val<
298  X, const Y, typename simplify_type<const Y>::SimpleType>::doit(Val);
299 }

318 inline typename cast_retty<X, std::unique_ptr<Y>>::ret_type
319  cast(std::unique_ptr<Y> &&Val)
320 {
321  assert(llvm::isa<X>(Val.get()) && "cast<Ty>() argument of incompatible type!");
322  using ret_type = typename cast_retty<X, std::unique_ptr<Y>>::ret_type;
